#include <iostream>
#include <vector>

using namespace std;

int n; // число вершин
int m; // число ребер
vector<int> *g; // граф
bool has_cycle = false;
char* used;
vector<int> sorted_list;

void dfs(int v) {
	used[v] = 1;
	for (size_t i = 0; i < g[v].size(); ++i) {
		int	to = g[v][i];
		if (!used[to])
			dfs(to);
		if (used[to] == 1)
		{
			has_cycle = true;
		}
	}
	used[v] = 2;
	sorted_list.push_back(v);
}

void print_list()
{
	for (auto i = 0; i < n; i++)
	{
		cout << sorted_list[i] + 1 << " ";
	}
	cout << endl;
}

void topological_sort() {
	for (int i = 0; i < n; ++i)
		used[i] = 0;
	sorted_list.clear();
	for (int i = 0; i < n; ++i)
	{
		if (!used[i])
		{
			dfs(i);
		}
	}
	reverse(sorted_list.begin(), sorted_list.end());
	if (!has_cycle) 
	{ 
		print_list(); 
	}
	else { 
		cout << -1 << endl; 
	}
}

void read_data()
{
	cin >> n >> m;
	used = new char[n] {0};
	g = new vector<int>[n];
	for (auto i = 0; i < m; i++)
	{
		int a = 0;
		int b = 0;
		cin >> a >> b;
		--a; --b;
		g[a].push_back(b);
	}
}

void print_g()
{
	for (size_t i = 0; i < n; i++)
	{
		cout << i + 1 << ":";
		for (size_t j = 0; j < g[i].size(); j++)
		{
			cout << g[i][j] + 1 << " ";
		}
		cout << endl;
		}
}

int main()
{
	read_data();
	//print_g();
	topological_sort();
	return 0;
}
